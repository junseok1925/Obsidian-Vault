void main() {
  Map<String, int> scores = {"철수": 75, "영희": 95, "민수": 88, "지영": 60};

  var data = scores.entries.toList();

  print(data);
  // 1번 방식
  for (var i in data) {
    late String grade;
    if (i.value >= 90) {
      grade = "우수";
    } else if (i.value >= 70) {
      grade = "보통";
    } else {
      grade = "미흡";
    }
    print("${i.key}, ${i.value}, $grade");
  }

  print("===================================");
  // 2번 방식
  scores.forEach((student, score) {
    late String grade;

    if (score >= 90) {
      grade = "우수";
    } else if (score >= 70) {
      grade = "보통";
    } else {
      grade = "미흡";
    }
    print("$student, $score, $grade");
  });
}

  print(sum);

  double avg = sum / process.length;
