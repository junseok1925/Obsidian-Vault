
```dart
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        body: Center(
          //Container를 Center로 감싸주어 해당 요소를 가운데로 정렬
          child: Container(
            padding: const EdgeInsets.only(left: 20, right: 20),
            decoration: BoxDecoration(
              // decoration 즉, 꾸미기 요소
              gradient: LinearGradient(
                colors: [
                  Color.fromARGB(255, 255, 59, 98).withOpacity(0.7),
                  Colors.blue,
                  Color.fromARGB(255, 255, 59, 98),
                ],
                begin: Alignment.topLeft,
                end: Alignment.bottomRight,
              ),
              borderRadius: BorderRadius.circular(10), // 모서리 둥근정도
              boxShadow: [
                BoxShadow(
                  // 요소 그림자
                  color: Color.fromARGB(255, 255, 59, 98).withOpacity(0.5),
                  spreadRadius: 5,
                  blurRadius: 7,
                  offset: Offset(0, 3), // changes position of shadow
                ),
              ],
            ),
            width: 200,
            height: 150,
            child: Center(
              child: Text('Container', style: TextStyle(color: Colors.white)),
            ),
          ),
        ),
      ),
    );
  }
}

```

- 기본적은 디지안 요소 ( padding, gradient, borderRadius)들은 React와 크게 다를게 없다.
- child 요소를 가운데 정렬하는데 Center로 감싸면 되는 것이 편리함
