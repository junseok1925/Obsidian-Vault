
### Text 위젯

```dart
import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

/// 앱의 진입점: 상단 AppBar + MotherForm(폼 화면) 출력
class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false, // 오른쪽 상단 "debug" 배너 숨기기
      home: Scaffold(
        // Scaffold: 화면의 기본 레이아웃 구조를 제공하는 위젯
        appBar: AppBar(
          title: Text('TextField Example'),
        ), // appBar 상단의 앱바(제목, 메뉴 등)
        body: Padding(
          // 메인 콘텐츠가 들어가는 공간
          padding: const EdgeInsets.all(
            16.0,
          ), // 위,아래,좌,우 16픽셀씩 패딩, const로 불멸 속성 부여하여 성능 최적화
          child: MotherForm(), // 메인 폼 위젯,
        ),
      ),
    );
  }
}

/// 부모 위젯: 상태 관리(StatefulWidget)
/// - TextEditingController로 입력을 관리
/// - 버튼을 누르면 입력값을 StyledText로 전달
class MotherForm extends StatefulWidget {
  @override
  _MotherFormState createState() => _MotherFormState();
  //MotherFormState 생성, MotherForm은 그냥 껍데기로 createState() 메서드로 상태 객체(_MyCustomFormState)를 생성
}

class _MotherFormState extends State<MotherForm> {
  //_MotherFormState라는 State 클래스를 정의, 이 State 클래스는 MotherForm이라는 StatefulWidget의 상태(UI + 데이터)를 관리한다.
  final TextEditingController myController = TextEditingController();
  // TextField의 입력값을 읽거나 조작하기 위한 컨트롤러 객체 생성, final로 불멸 속성을 부여하여 변수 자체를 다른 것으로 바꿀 수는 없으나, 내부 값은 계속 바뀔 수 있다.
  String displayedText = ""; // 버튼 눌렀을 때 표시할 텍스트를 담는 변수 선언

  @override
  Widget build(BuildContext context) {
    // 위젯이 화면에 어떻게 보일지 정의하는 함수
    return Column(
      // Column 위젯으로 세로로 정렬하는 레이아웃 컨테이너 생성
      children: [
        // 입력된 텍스트를 출력하는 위젯
        StyledText(text: displayedText), // 자식 위젯1 : StyledText
        SizedBox(height: 100), // 위/아래 여백을 주는 위젯
        // 텍스트 입력 필드
        StyledTextField(controller: myController), // 자식 위젯2 : StyledTextField

        SizedBox(height: 100), // 위/아래 여백을 주는 위젯
        // 버튼: 입력된 텍스트를 위쪽 StyledText로 전달
        ElevatedButton(
          onPressed: () {
            setState(() {
              displayedText = myController.text; // 입력된 텍스트를 displayedText에 저장
              myController.clear(); // 입력창 비우기
            });
          },
          child: Text('입력'),
        ),
      ],
    );
  }

  @override
  void dispose() {
    myController.dispose(); // 메모리 누수 방지
    super.dispose();
  }

  // dispose()는 위젯이 화면에서 사라질 때 호출되는 정리 메서드
  // controller, AnimationController, Stream 등 리소스 자원을 쓰는 객체는 dispose()를 해줘야 메모리 누수가 안생기며 앱 성능을 안전하게 유지할 수 있다.
}

/// 자식 위젯 1: 출력 전용 텍스트
class StyledText extends StatelessWidget {
  //StyledText라는 StatelessWidget을 정의
  final String text;
  // 외부엣 전달받은 값을 저장하는 변수, text라는 문자열을 부모 위젯이 전달하면, 이 자식 위젯에서 사용이 가능
  const StyledText({super.key, required this.text});
  // super.key: 본 위젯이 고유하게 식별될 수 있게 Flutter가 내부적으로 위젯 트리 비교용으로 사용하는 식별자
  // required this.text : text는 반드시 전달되어야 하는 필수 매개변수

  @override
  Widget build(BuildContext context) {
    return Text(
      '입력된 텍스트: $text', // 부모로부터 전달받은 text 변수 출력
      style: TextStyle(
        fontSize: 20,
        fontWeight: FontWeight.bold,
        color: Colors.blue,
        fontStyle: FontStyle.italic,
        letterSpacing: 2.0,
        wordSpacing: 5.0,
        decoration: TextDecoration.underline,
        fontFamily: 'Roboto',
      ),
    );
  }
}

/// 자식 위젯 2: 입력 필드
class StyledTextField extends StatelessWidget {
  // StatelessWidget를 상속받는 위젯 StyledTextField 생성
  final TextEditingController controller;
  // TextEditingController라는 컨트롤러를 필수로 전달 받아야 함(필수매개변수)

  const StyledTextField({super.key, required this.controller});

  @override
  Widget build(BuildContext context) {
    return TextField(
      controller: controller,
      maxLength: 20,
      decoration: InputDecoration(
        labelText: '입력하시오',
        border: OutlineInputBorder(),
      ),
    );
  }
}


```


- text위젯에 대해 공부한 내용을 한 화면에서 보고싶어 한 번 만들어 봤다. 사용자에게 텍스트를 입력 받아 "입력" 버튼을 누르면 텍스트위젯으로 표시되도록 StatelessWidget과 StatefulWidget을 이용해서 만들어 봤다.
- StatelessWidget과 StatefulWidget에 대해 좀 더 익숙해졌다.
- 부모와 자식 위젯의 관계가 좀 어렵지만 계속 보다보면 익숙해진다.
- 보통 이렇게 한 코드에 부모, 자식 위젯을 코딩하는 경우는 없고 파일별로 나눠서 만든다.
