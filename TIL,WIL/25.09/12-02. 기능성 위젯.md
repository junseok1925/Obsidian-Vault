
### Text 위젯

```dart
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false, // 오른쪽 상단에 debug 배너를 숨긴다
      home: Scaffold(
        body: Center(
          // 가운데 정렬
          child: Text(
            '스타일적용 텍스트',
            style: TextStyle(
              //텍스트 스타일 적용
              fontSize: 20, // 텍스트 사이즈
              fontWeight: FontWeight.bold, // 텍스트 굵기
              color: Colors.blue, // 텍스트 컬러
              fontStyle: FontStyle.italic, // 폰트 설정
              letterSpacing: 10.0, // 글자 간격 조정
              wordSpacing: 10.0, // 단어 간격 조정
              decoration: TextDecoration.underline, // 텍스트 밑줄
              fontFamily: 'Roboto', // 사용할 폰트 패밀리 이름 ( 프로젝트에 폰트가 있어야 함)
            ),
          ),
        ),
      ),
    );
  }
}

```
