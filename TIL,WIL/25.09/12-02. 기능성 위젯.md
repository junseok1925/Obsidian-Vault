// 별도의 StatelessWidget으로 텍스트 스타일 정의
class StyledText extends StatelessWidget {
  final String text;
  const StyledText({super.key, required this.text});
  @override
  Widget build(BuildContext context) {
    return Text(
      '입력된 텍스트: $text',
      style: TextStyle(
        fontSize: 20, // 텍스트 사이즈
        fontWeight: FontWeight.bold, // 텍스트 굵기
        color: Colors.blue, // 텍스트 컬러
        fontStyle: FontStyle.italic, // 폰트 설정
        letterSpacing: 10.0, // 글자 간격 조정
        wordSpacing: 10.0, // 단어 간격 조정
        decoration: TextDecoration.underline, // 텍스트 밑줄
        fontFamily: 'Roboto', // 사용할 폰트 패밀리 이름
      ),
    );
  }
}

// 사용자 입력 inputText 위젯 생성
class StyledTextField extends StatelessWidget {
  final TextEditingController controller;

  const StyledTextField({super.key, required this.controller});

  @override
  Widget build(BuildContext context) {
    return TextField(
      controller: controller,
      maxLength: 20,
      decoration: InputDecoration(
        labelText: '입력하시오',
        border: OutlineInputBorder(),
      ),
    );
  }
}
