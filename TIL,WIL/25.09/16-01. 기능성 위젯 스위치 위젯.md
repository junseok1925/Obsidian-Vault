

```dart
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatefulWidget {
  const MyApp({super.key});

  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  var switchState = false; // Switch 위젯의 on/off 상태를 저장하는 역할
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        body: Center(
          child: Switch(
            value: switchState,
            onChanged: (bool newValue) {
              //onChanged : setState()로 state를 newValue로 바꾸면 -> flutter가 build를 다시 실행 -> UI가 업데이트 됨
              // 토글 이벤트 처리
              setState(() {
                switchState = newValue;
              });
            },
          ),
        ),
      ),
    );
  }
}

```

- switch의 상태에 따라 UI가 업데이트 되므로 statefulWidget 사용
