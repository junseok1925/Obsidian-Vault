
AppBar, Container, SizeBox, Row Colum 위젯을 합쳐 간단하게 볼 수 있는 예제를 만들어 봤다.

```dart
import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        // AppBar : Flutter화면 상단에 표시되는 앱의 제목, 버튼, 액션... 등 담는 영역을 제공하는 위젯임. 쉽게 말해 앱의 "header"의 역할
        appBar: AppBar(
          title: Text("Flutter UI 연습"),
          backgroundColor: Colors.blue, // 배경색
          foregroundColor: Colors.red, // 텍스트, 아이콘 색
          elevation: 4, // 그림자
          centerTitle: true, // 제목 중앙 배치
          leading: IconButton(
            icon: Icon(Icons.menu),
            onPressed: () {
              print("메뉴 버튼");
            },
          ),
          actions: [
            IconButton(
              icon: Icon(Icons.search),
              onPressed: () {
                print("검색 버튼");
              },
            ),
            IconButton(
              icon: Icon(Icons.settings),
              onPressed: () {
                print("설정 버튼");
              },
            ),
          ],
        ),

        //body : 앱의 주요 콘텐츠를 표시하는 영역을 제공하는 위젯임. 쉽게 말해 앱의 "main content"의 역할
        body: SingleChildScrollView(
          //child : 화면에 표시할 위젯을 지정하는 속성임. body는 하나의 위젯만 가질 수 있기 때문에, 여러 위젯을 포함하려면 Column, Row, Stack 등과 같은 레이아웃 위젯을 사용해야 함.
          child: Padding(
            padding: const EdgeInsets.all(20.0),
            // 자식 요소를 세로 방향으로 배치하기 위해 Column 위젯 사용
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                // Row 예제 가로로 텍스트 배치
                Row(
                  mainAxisAlignment: MainAxisAlignment.spaceEvenly,
                  children: [Text("왼쪽"), Text("가운데"), Text("오른쪽")],
                ),

                // 위젯을 구분하기 위한 위젯
                SizedBox(height: 50),

                // 아래 텍스트
                Text("아래에 위치한 텍스트"),

                // 위젯을 구분하기 위한 위젯
                SizedBox(height: 50),

                // Container + Text
                Container(
                  width: 200,
                  height: 100,
                  color: Colors.blue,
                  child: Center(
                    child: Text(
                      "container 안의 텍스트",
                      style: TextStyle(fontSize: 24, color: Colors.white),
                    ),
                  ),
                ),
                SizedBox(height: 20),

                // 4️⃣ Image + Icon
                Image.network(
                  "https://via.placeholder.com/150",
                  width: 150,
                  height: 150,
                ),
                SizedBox(height: 10),
                Icon(Icons.favorite, size: 50, color: Colors.red),

                //버튼 생성
                ElevatedButton(
                  onPressed: () {
                    print("버튼 눌림");
                  },
                  child: Text('누르셈'),
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }
}


```
